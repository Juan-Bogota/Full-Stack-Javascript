        // Operadores de incremento

        a = 5;

        console.log(a);   // 5
        console.log(a++); // 5

        console.log(a);   // 6
        console.log(++a); // 7

        console.log(a);   // 7
        console.log(a--); // 7

        console.log(a);   // 6
        console.log(--a); // 5

        // Operadores de Asignacion

        a = 10;

        a += 3;
        console.log('Suma de 10 + 3 = ' + a); 

        a -= 3;
        console.log('Resta de 13 - 3 = ' + a); 

        a *= 3;
